# app.py
@app.post("/webhook/mercury")
def mercury_webhook():
    event = verify_mercury_webhook(request)
    intent_id = event["data"]["id"]
    status = event["data"]["status"]   # e.g., 'paid'
    pay = Payment.query.filter_by(intent_id=intent_id, provider="mercury").first_or_404()
    if status == "paid":
        pay.status = "paid"
        if pay.booking_id:
            b = Booking.query.get(pay.booking_id); b.status = "paid"
        if pay.pass_id:
            p = Pass.query.get(pay.pass_id); p.status = "active"
        db.session.commit()
    return "ok"

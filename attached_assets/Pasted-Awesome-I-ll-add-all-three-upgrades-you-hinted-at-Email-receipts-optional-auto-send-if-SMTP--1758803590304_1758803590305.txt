1) Add helpers to main.py (near other imports / utils)

import smtplib
from email.mime.text import MIMEText
from email.utils import formataddr

def email_enabled():
    return all([
        os.getenv("SMTP_HOST"),
        os.getenv("SMTP_PORT"),
        os.getenv("SMTP_USER"),
        os.getenv("SMTP_PASS"),
        os.getenv("FROM_EMAIL"),
        os.getenv("FROM_NAME"),
    ])

def send_receipt_email(booking: Booking):
    if not email_enabled(): 
        return
    host = os.getenv("SMTP_HOST")
    port = int(os.getenv("SMTP_PORT", "587"))
    user = os.getenv("SMTP_USER")
    pwd  = os.getenv("SMTP_PASS")
    from_email = os.getenv("FROM_EMAIL")
    from_name  = os.getenv("FROM_NAME")
    subject = f"Your EasyDesk booking #{booking.id}"
    body = f"""
Hi {booking.name},

Your booking is confirmed.

When: {booking.start_dt:%A, %b %d %Y %I:%M %p} – {booking.end_dt:%I:%M %p}
Resource: {booking.resource.name}
Status: {booking.status.upper()}
Amount: {as_money(booking.amount_cents)}

Add to calendar:
{BASE_URL}/ics/{booking.id}

Need help? Reply to this email.

— City Discoverer EasyDesk
"""
    msg = MIMEText(body)
    msg["From"] = formataddr((from_name, from_email))
    msg["To"]   = booking.email
    msg["Subject"] = subject

    with smtplib.SMTP(host, port) as s:
        s.starttls()
        s.login(user, pwd)
        s.sendmail(from_email, [booking.email], msg.as_string())

3) Add .ics calendar download route

@app.get("/ics/<int:bid>")
def ics(bid):
    b = Booking.query.get_or_404(bid)
    uid = f"booking-{b.id}@easydesk"
    def fmt(dt_obj): return dt_obj.strftime("%Y%m%dT%H%M%S")
    ics_text = f"""BEGIN:VCALENDAR
VERSION:2.0
PRODID:-//City Discoverer EasyDesk//EN
BEGIN:VEVENT
UID:{uid}
DTSTAMP:{fmt(dt.datetime.utcnow())}Z
DTSTART:{fmt(b.start_dt)}
DTEND:{fmt(b.end_dt)}
SUMMARY:EasyDesk – {b.resource.name}
DESCRIPTION:Booking #{b.id} – {as_money(b.amount_cents)} – Status: {b.status}
LOCATION:50 Stately St, Suite 2, Wiley Ford WV 26767
END:VEVENT
END:VCALENDAR
"""
    return send_file(
        io.BytesIO(ics_text.encode("utf-8")),
        mimetype="text/calendar",
        as_attachment=True,
        download_name=f"easydesk-booking-{b.id}.ics"
    )

4) Day-view availability grid on the booking page

4a) Add a small query helper in main.py

def day_bookings(date_str):
    # returns bookings grouped by resource for that day
    y, m, d = map(int, date_str.split("-"))
    start = dt.datetime(y, m, d, 0, 0)
    end   = start + dt.timedelta(days=1)
    rows = (Booking.query
            .filter(Booking.start_dt < end,
                    Booking.end_dt > start,
                    Booking.status.in_(["reserved","paid","checked_in","free"]))
            .order_by(Booking.resource_id, Booking.start_dt)
            .all())
    grouped = {}
    for b in rows:
        grouped.setdefault(b.resource_id, []).append(b)
    return grouped
